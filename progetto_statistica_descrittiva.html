gini_index=function(x){
  ni=table(x)
  fi=ni/length(x)
  fi2=fi^2
  j=length(table(x))
  
  gini=1-sum(fi2)
  gini.normalizzato=gini/((j-1)/j)
  
  return(gini.normalizzato)
}
gini_index(dati$sales_cl)
